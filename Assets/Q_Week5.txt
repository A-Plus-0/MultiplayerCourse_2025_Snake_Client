private Material NewColorMaterial(Material mat)
{ 
	Material result = new Material(mat);
	Color.RGBToHSV(result.color, out float h, out float s, out float v);
	result.color = Color.HSVToRGB(h, s, v * 0.9f);
	return result;
}
